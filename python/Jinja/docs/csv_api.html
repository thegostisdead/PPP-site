import csv
#

parametres : collum_name : la collone a lire csv_file : le fichier csv resultat : le text des collones 

def readCollum(collum_name,csv_filename):
# 
    with open(csv_filename, mode='r') as csv_file:
        res = []
        csv_reader = csv.DictReader(csv_file)
        line_count = 0
        for row in csv_reader:
            if line_count == 0:
                line_count += 1
            res.append(row[collum_name])
            line_count += 1
        return res
